sentence = "The whole Christmas tree."

#Find the longest word 
max_len = -1
for word in sentence.split(): 
    if len(word) > max_len: 
        max_len = len(word) 
        longest = word 
        longest_length = len(longest)

#Create tree

    counter = 1
    for letter in x:
        a = (letter *counter)*2
        print(a.center(longest_length*2))
        counter +=1

s = 'OMGOOOOOMMMMMMGGG'

#Find change_idx
change_idx = []
for index in range(0,len(s)-1): 
    if s[index] != s[index + 1]:
        change_idx.append(index + 1)
    else: 
        continue

#Find longest sequence
if change_idx: 
    change_idx.insert(0,0) #include first number
    change_idx.append(len(s)) #include last number
    longest = max([change_idx[i+1]-change_idx[i] for i in range(len(change_idx)-1)])
    change_idx = change_idx[1:-1]
else: 
    print("Change_idx is empty.")

f'The longest sequence is {longest}.'
